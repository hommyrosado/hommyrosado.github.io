Example script to generate barcharts:
import pandas as pd
import json
import matplotlib.pyplot as plt 

# Use pandas to read in the csv file.  It uses relative the
# relative path
df = pd.read_csv('cpb_zeek_http_mission_ocx.csv')
# Print the first 5 entries of the data set
df.head()
# We are interested in 2 pieces of information at first: url and source ip
url_domain_reverse_list = []
url_original_list = []
url_domain_list = []
for i, j in df.iterrows():
    tmp = j['url']
    if isinstance(tmp, str):
        tmp_json = json.loads(tmp)
        url_domain_reverse_list.append(tmp_json['domain_reverse'])
        url_original_list.append(tmp_json['original'])
        url_domain_list.append(tmp_json['domain'])

# Get unique list
unique_url_domain_reverse_list = list(set(url_domain_reverse_list))
url_domain_reverse_count = []
unique_url_original_list = list(set(url_original_list))
url_original_count = []
unique_url_domain_list = list(set(url_domain_list))
url_domain_count = []
# Gather the count of each url from the total data set
for i in unique_url_domain_reverse_list:
    tmp_count = url_domain_reverse_list.count(i)
    url_domain_reverse_count.append(tmp_count)

for i in unique_url_original_list:
    tmp_count = url_original_list.count(i)
    url_original_count.append(tmp_count)
for i in unique_url_domain_list:
    tmp_count = url_domain_list.count(i)
    url_domain_count.append(tmp_count)
# Start plotting bar charts
fig = plt.figure()
# creating the bar plot
plt.barh(unique_url_domain_reverse_list[0:6], url_domain_reverse_count[0:6])
plt.xlabel("Visits")
plt.ylabel("Domain Reverse List")
plt.show()
